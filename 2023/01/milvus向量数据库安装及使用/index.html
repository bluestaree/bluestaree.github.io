Milvus向量数据库安装及使用

​ 项目需求包含以图搜人功能，查询目前相关解决方案，milvus热度较高，决定基于milvus进行以图搜人开发，原理是结合 虹软4.0 提取图片特征值，将特征值转为256位特征值向量， 存入milvus进行距离计算，以此估计图片相似度。通过百万数据测试，milvus距离大于0.55的，虹软匹配结果相似度在80%左右

milvus环境搭建

mivlus官方文档：https://milvus.io/docs/v2.1.x

这里采用docker方式部署,注意文档中的最低版本要求

  • Docker 19.03 or later
  • Docker Compose 1.25.1 or later

docker-cpmpoese方式部署单机版本

参考文档，首页拉取 docker-cpmpoese 文件

wget https://github.com/milvus-io/milvus/releases/download/v2.1.4/milvus-standalone-docker-compose.yml -O docker-compose.yml

使用部署命令 docker-compose up -d

如需停止所有相关组件： docker-compose stop

如需启动所有相关组件： docker-compose start

如需停止并删除容器： docker-compose down

Milvus连接工厂

@Component
public class MilvusPoolFactory extends BasePooledObjectFactory<MilvusServiceClient> {

    public static MilvusPoolFactory milvusPoolFactory;

    @Autowired
    private MilvusConfig milvusConfig;

    @PostConstruct
    public void init() {
        milvusPoolFactory = this;
        milvusPoolFactory.milvusConfig = milvusConfig;
        // 链接预热
        MilvusServiceClient client = milvusPoolFactory.create();
        client.hasCollection(HasCollectionParam.newBuilder()
                .withCollectionName("test")
                .build());
        client.close();
    }

    @Override
    public MilvusServiceClient create() {
        ConnectParam connectParam = ConnectParam.newBuilder()
                .withHost(milvusPoolFactory.milvusConfig.getHost())
                .withPort(milvusPoolFactory.milvusConfig.getPort())
                .build();
        MilvusServiceClient client = new MilvusServiceClient(connectParam);
        return client;
    }

    @Override
    public PooledObject<MilvusServiceClient> wrap(MilvusServiceClient milvusServiceClient) {
        return new DefaultPooledObject<>(milvusServiceClient);

    }
}

Milvus配置文件

@Data
@Component
@ConfigurationProperties(prefix = "aiot.milvus")
public class MilvusConfig {

    /**
     * milvus所在服务器地址
     */
    private String host;

    /**
     * milvus所在服务器端口
     */
    private Integer port;

    /**
     * 以图搜人查询因子 越低查询得数据就越多，计算方式 前端传递相似度 * 查询因子 = milvus - score
     */
    private BigDecimal factor;

    @Bean("milvusServiceClientGenericObjectPool")
    public GenericObjectPool milvusServiceClientGenericObjectPool() {
        MilvusPoolFactory milvusPoolFactory = new MilvusPoolFactory();
        // 对象池配置
        GenericObjectPoolConfig<FaceEngine> objectPoolConfig = new GenericObjectPoolConfig<>();
        objectPoolConfig.setMaxTotal(8);
        objectPoolConfig.setJmxEnabled(false);

        AbandonedConfig abandonedConfig = new AbandonedConfig();

        // 在Maintenance的时候检查是否有泄漏
        abandonedConfig.setRemoveAbandonedOnMaintenance(true);

        // borrow 的时候检查泄漏
        abandonedConfig.setRemoveAbandonedOnBorrow(true);

        // 如果一个对象borrow之后10秒还没有返还给pool，认为是泄漏的对象
        abandonedConfig.setRemoveAbandonedTimeout(5);

        // 对象池
        GenericObjectPool<MilvusServiceClient> milvusServiceClientGenericObjectPool = new GenericObjectPool(milvusPoolFactory, objectPoolConfig);
        milvusServiceClientGenericObjectPool.setAbandonedConfig(abandonedConfig);
        // 5秒运行一次维护任务
        milvusServiceClientGenericObjectPool.setTimeBetweenEvictionRunsMillis(5000);
        return milvusServiceClientGenericObjectPool;
    }
}

@Data
@Component
@Slf4j
public class MilvusOperateUtils {

    @Autowired
    private MilvusConfig milvusConfig;

    @Autowired
    @Qualifier(value = "milvusServiceClientGenericObjectPool")
    private GenericObjectPool<MilvusServiceClient> milvusServiceClientGenericObjectPool;

    /**
     * 创建集合（库）
     *
     * @param collectionName 集合名
     * @param milvusServiceClient 操作客户端对象
     * @return void
     */
    public boolean createCollection(String collectionName, MilvusServiceClient milvusServiceClient) {
        // 参数校验
        if(StrUtil.isEmpty(collectionName)) {
            log.error("未输入集合名称");
            return false;
        }
        boolean needReturn = false;
        try {
            if(milvusServiceClient == null) {
                // 通过对象池管理对象
                milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();
                needReturn = true;
            }

            CreateCollectionParam createCollectionReq = null;

            // 判断是否存在集合信息
            boolean hasCollection = hasCollection(collectionName, milvusServiceClient);
            if (hasCollection) {
                // 集合已创建
                return true;
            }

            if(StrUtil.equals(collectionName, MilvusConstants.COLLECTION_FACE_NAME)) {
                FieldType collectionId = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceArchiveField.COLLECTION_ID)
                        .withDescription("集合主键id")
                        .withDataType(DataType.Int64)
                        .withPrimaryKey(true)
                        .withAutoID(false)
                        .build();
                FieldType memberId = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceArchiveField.MEMBER_ID)
                        .withDescription("用户id")
                        .withDataType(DataType.Int64)
                        .build();
                FieldType timestamp = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceArchiveField.TIMESTAMP)
                        .withDescription("时间戳")
                        .withDataType(DataType.Int64)
                        .build();
                FieldType faceFeature = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceArchiveField.FACE_FEATURE)
                        .withDescription("人脸特征向量")
                        .withDataType(DataType.FloatVector)
                        .withDimension(MilvusConstants.FEATURE_DIM)
                        .build();
                createCollectionReq = CreateCollectionParam.newBuilder()
                        .withCollectionName(collectionName)
                        .withDescription("人脸特征向量库")
                        .withShardsNum(MilvusConstants.SHARDS_NUM)
                        .addFieldType(collectionId)
                        .addFieldType(memberId)
                        .addFieldType(timestamp)
                        .addFieldType(faceFeature)
                        .build();
            } else if(collectionName.startsWith(MilvusConstants.COLLECTION_LOG_PREFIX)) {
                FieldType collectionId = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.COLLECTION_ID)
                        .withDescription("集合主键id")
                        .withDataType(DataType.Int64)
                        .withPrimaryKey(true)
                        .withAutoID(false)
                        .build();
                FieldType logId = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.LOG_ID)
                        .withDescription("日志id")
                        .withDataType(DataType.Int64)
                        .build();
                FieldType logType = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.LOG_TYPE)
                        .withDescription("日志类型")
                        .withDataType(DataType.Int32)
                        .build();
                FieldType deviceId = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.DEVICE_ID)
                        .withDescription("设备id")
                        .withDataType(DataType.Int64)
                        .build();
                FieldType timestamp = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.TIMESTAMP)
                        .withDescription("时间戳")
                        .withDataType(DataType.Int64)
                        .build();
                FieldType faceFeature = FieldType.newBuilder()
                        .withName(MilvusConstants.FaceSnapshotField.FACE_FEATURE)
                        .withDescription("图片特征向量")
                        .withDataType(DataType.FloatVector)
                        .withDimension(MilvusConstants.FEATURE_DIM)
                        .build();
                createCollectionReq = CreateCollectionParam.newBuilder()
                        .withCollectionName(collectionName)
                        .withDescription("日志抓拍特征向量库")
                        .withShardsNum(MilvusConstants.SHARDS_NUM)
                        .addFieldType(collectionId)
                        .addFieldType(logId)
                        .addFieldType(logType)
                        .addFieldType(deviceId)
                        .addFieldType(timestamp)
                        .addFieldType(faceFeature)
                        .build();
            } else {
                log.error("预设集合信息不存在");
                return false;
            }

            R<RpcStatus> result = milvusServiceClient.createCollection(createCollectionReq);
            if (result.getStatus() != 0) {
                log.error("Milvus创建集合失败 集合:{}，结果：{}", collectionName, result);
                return false;
            }
            // 创建索引
            createIndex(collectionName, MilvusConstants.FaceArchiveField.FACE_FEATURE, "{\"nlist\":2048}", milvusServiceClient);
            return true;
        } catch (Exception e) {
            log.error("创建集合异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null && needReturn) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
        return false;
    }

    /**
     * 创建索引
     *
     * @param collectionName 集合名
     * @param milvusServiceClient 操作客户端对象
     * @return void
     */
    public boolean createIndex(String collectionName, String indexFileName, String extraParam, MilvusServiceClient milvusServiceClient) {
        boolean needReturn = false;
        try {
            if(milvusServiceClient == null) {
                // 通过对象池管理对象
                milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();
                needReturn = true;
            }
            R<RpcStatus> result = milvusServiceClient.createIndex(
                    CreateIndexParam.newBuilder()
                            // 集合名称
                            .withCollectionName(collectionName)
                            // 索引字段名称
                            .withFieldName(indexFileName)
                            // 索引类型
                            .withIndexType(IndexType.IVF_FLAT)
                            .withMetricType(MetricType.IP)
                            // nlist 建议值为 4 × sqrt(n)，其中 n 指 segment 最多包含的 entity 条数
                            .withExtraParam(extraParam)
                            .withSyncMode(Boolean.FALSE)
                            .build());
            if (result.getStatus() != 0) {
                log.error("Milvus创建集合索失败 集合:{}，结果：{}", collectionName, result);
                return false;
            }
            return true;
        } catch (Exception e) {
            log.error("创建集合索引异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null && needReturn) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
        return false;
    }

    /**
     * 创建分区
     *
     * @param collectionName 集合名
     * @param milvusServiceClient 操作客户端对象
     * @return void
     */
    public void createPartition(String collectionName, MilvusServiceClient milvusServiceClient) {
        boolean needReturn = false;
        try {
            if(milvusServiceClient == null) {
                // 通过对象池管理对象
                milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();
                needReturn = true;
            }

            // 根据定义分区数， 循环创建
            for (int i = 0; i < MilvusConstants.PARTITION_NUM; i++) {
                String partitionName = MilvusConstants.PARTITION_PREFIX + i;
                R<RpcStatus> result = milvusServiceClient.createPartition(
                        CreatePartitionParam.newBuilder()
                                .withCollectionName(collectionName)
                                .withPartitionName(partitionName)
                                .build());
                if (result.getStatus() != 0) {
                    log.error("Milvus创建集合分区失败 集合:{}，分区:{}，结果：{}", collectionName, partitionName, result);
                }
            }
        } catch (Exception e) {
            log.error("创建分区异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null && needReturn) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
    }

    /**
     * 判断是否存在集合
     *
     * @param collectionName 集合名
     * @param milvusServiceClient 操作客户端对象
     * @return void
     */
    public boolean hasCollection(String collectionName, MilvusServiceClient milvusServiceClient) {
        boolean needReturn = false;
        try {
            if(milvusServiceClient == null) {
                // 通过对象池管理对象
                milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();
                needReturn = true;
            }
            // 判断是否存在集合信息
            R<Boolean> result = milvusServiceClient.hasCollection(
                    HasCollectionParam.newBuilder()
                            .withCollectionName(collectionName)
                            .build());
            if (result.getStatus() != 0) {
                log.error("Milvus判断是否存在集合失败,原因 :{} ", result);
                return false;
            }
            if (result.getStatus() == 0 && Boolean.TRUE.booleanValue() == result.getData()) {
                // 集合已创建
                return true;
            }
        } catch (Exception e) {
            log.error("判断集合是否存在异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null && needReturn) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
        return false;
    }

    /**
     * 插入数据
     *
     * @param faceArchiveDto 传输数据
     * @return void
     */
    public long insert(MilvusFaceArchiveDTO faceArchiveDto) {
        MilvusServiceClient milvusServiceClient = null;
        try {
            // 获取插入集合信息
            String collectionName = MilvusConstants.getCollectionName(faceArchiveDto.getCollectionPrefix(), new Date());

            // 获取插入集合分区信息
            String partitionName = MilvusConstants.getPartitionName(faceArchiveDto.getTenantId());

            // 参数校验
            if(StrUtil.isEmpty(collectionName) || StrUtil.isEmpty(partitionName)) {
                log.error("插入数据失败： 无法获取集合名称或分区名称");
                return 0;
            }

            // 通过对象池管理对象
            milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();

            // 组装字段数据
            List<InsertParam.Field> fields = new ArrayList<>();
            if(StrUtil.equals(collectionName, MilvusConstants.COLLECTION_FACE_NAME)) {
                fields.add(new InsertParam.Field(MilvusConstants.FaceArchiveField.COLLECTION_ID, faceArchiveDto.getCollectionIdList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceArchiveField.MEMBER_ID, faceArchiveDto.getMemberIdList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceArchiveField.TIMESTAMP,  faceArchiveDto.getTimestampList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceArchiveField.FACE_FEATURE,  faceArchiveDto.getFeatureList()));
            } else if(collectionName.startsWith(MilvusConstants.COLLECTION_LOG_PREFIX)) {
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.COLLECTION_ID, faceArchiveDto.getCollectionIdList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.LOG_ID, faceArchiveDto.getLogIdList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.LOG_TYPE, faceArchiveDto.getLogTypeList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.DEVICE_ID,  faceArchiveDto.getDeviceIdList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.TIMESTAMP,  faceArchiveDto.getTimestampList()));
                fields.add(new InsertParam.Field(MilvusConstants.FaceSnapshotField.FACE_FEATURE,  faceArchiveDto.getFeatureList()));
            } else {
                log.error("预设集合信息不存在");
                return 0;
            }

            // 判断是否存在集合信息
            boolean hasCollection = hasCollection(collectionName, milvusServiceClient);
            if (!hasCollection) {
                // 创建集合
                boolean b = createCollection(collectionName, milvusServiceClient);
                if(b) {
                    // 创建分区
                    createPartition(collectionName, milvusServiceClient);
                } else{
                    log.error("Milvus数据上传失败,未成功创建集合");
                    return 0;
                }
            }

            InsertParam insertParam = InsertParam.newBuilder()
                    .withCollectionName(collectionName)
                    .withPartitionName(partitionName)
                    .withFields(fields)
                    .build();
            R<MutationResult> insertResult = milvusServiceClient.insert(insertParam);

            if (insertResult.getStatus() == 0) {
//                log.info("Milvus数据上传成功,结果 :{} ", insertResult.getData().getIDs().getIntId());
                return insertResult.getData().getIDs().getIntId().getData(0);
            } else {
                log.error("Milvus数据上传失败,原因 :{} ", insertResult);
            }
        } catch (Exception e) {
            log.error("插入集合数据异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
        return 0;
    }

    /**
     * 根据向量搜索数据
     *
     * @param faceSearchVO 查询信息
     * @param builder 查询条件
     * @param milvusServiceClient 操作客户端对象
     * @return void
     */
    public List<MilvusSearchResultVO> searchByFeature(MilvusFaceSearchVO faceSearchVO, SearchParam.Builder builder, MilvusServiceClient milvusServiceClient) {
        boolean needReturn = false;
        List<MilvusSearchResultVO> resultVOS = new ArrayList<>();
        try {
            if(milvusServiceClient == null) {
                // 通过对象池管理对象
                milvusServiceClient = milvusServiceClientGenericObjectPool.borrowObject();
                needReturn = true;
            }

            String collectionName = faceSearchVO.getCollectionName();

            // 参数校验
            if(StrUtil.isEmpty(collectionName)) {
                return resultVOS;
            }

            if(builder == null && (builder = getSearchBuilder(faceSearchVO)) == null) {
                return resultVOS;
            }

            // 查询前加载集合进入缓存
            loadingCollection(collectionName, milvusServiceClient);

            // 发起搜索请求
            R<SearchResults> respSearch = milvusServiceClient.search(builder.build());

            // 组装结果数据
            if (respSearch.getStatus() == 0 && !StrUtil.equals(respSearch.getData().getStatus().getReason(), MilvusConstants.EMPTY_REASON)){
                SearchResultsWrapper wrapperSearch = new SearchResultsWrapper(respSearch.getData().getResults());
                // 获取查询结果
                List<SearchResultsWrapper.IDScore> idScore = wrapperSearch.getIDScore(0);
//                log.info("查询结果：{}", idScore);
            }
            // 释放集合
//            releaseCollection(collectionName, milvusServiceClient);
        } catch (Exception e) {
            log.error("数据查询异常 :{}", e);
        } finally {
            // 回收对象到对象池
            if (milvusServiceClient != null && needReturn) {
                milvusServiceClientGenericObjectPool.returnObject(milvusServiceClient);
            }
        }
        return resultVOS;
    }

    /**
     * 构建milvus查询条件
     *
     * @param faceSearchVO
     * @return Builder
     */
    private SearchParam.Builder getSearchBuilder(MilvusFaceSearchVO faceSearchVO) {
        boolean timeScope = false;
        String collectionName = faceSearchVO.getCollectionName();

        // 定义输出字段
        List<String> searchOutputFields = new ArrayList<>();
        if(StrUtil.equals(collectionName, MilvusConstants.COLLECTION_FACE_NAME)) {
            searchOutputFields.add(MilvusConstants.FaceArchiveField.COLLECTION_ID);
            searchOutputFields.add(MilvusConstants.FaceArchiveField.MEMBER_ID);
        } else if(collectionName.startsWith(MilvusConstants.COLLECTION_LOG_PREFIX)) {
            searchOutputFields.add(MilvusConstants.FaceSnapshotField.COLLECTION_ID);
            searchOutputFields.add(MilvusConstants.FaceSnapshotField.LOG_ID);
            searchOutputFields.add(MilvusConstants.FaceSnapshotField.LOG_TYPE);
            searchOutputFields.add(MilvusConstants.FaceSnapshotField.DEVICE_ID);
            searchOutputFields.add(MilvusConstants.FaceSnapshotField.TIMESTAMP);
            timeScope = true;
        } else {
            log.info("查询失败：集合不存在");
            return null;
        }

        List<String> partitionList = new ArrayList<>();
        // 获取租户id
        String tenantId = faceSearchVO.getTenantId();
        if (StrUtil.isNotEmpty(tenantId)) {
            // 计算当前租户查询分区
            String partitionName = MilvusConstants.getPartitionName(tenantId);
            partitionList.add(partitionName);
        }

        // 构建查询参数
        SearchParam.Builder builder = SearchParam.newBuilder()
                // 查询集合
                .withCollectionName(collectionName)
                // 查询分区
                .withPartitionNames(partitionList)
                // 计算方式
                .withMetricType(MetricType.IP)
                // 输出字段
                .withOutFields(searchOutputFields)
                // 查询数据量
                .withTopK(faceSearchVO.getTopK())
                // 查询向量
                .withVectors(faceSearchVO.getSearchVectors())
                // 向量字段
                .withVectorFieldName(MilvusConstants.FaceArchiveField.FACE_FEATURE)
                // 查询参数
                .withParams(faceSearchVO.getParam());

        // 构建查询条件
        StringBuffer sb = new StringBuffer();
        // 设备id
        if (StrUtil.isNotEmpty(faceSearchVO.getDeviceId())) {
            String[] split = faceSearchVO.getDeviceId().split(StrUtil.COMMA);
            sb.append(" ( ");
            for (int i = 0; i < split.length; i++) {
                sb.append(MilvusConstants.FaceSnapshotField.DEVICE_ID);
                sb.append(" == ");
                sb.append(split[i]);
                if(i != split.length - 1) {
                    sb.append(" || ");
                }
            }
            sb.append(" ) ");
            sb.append(" && ");
        }
        // 搜索范围
        Integer searchType = faceSearchVO.getSearchType();
        if(searchType != null) {
            sb.append(MilvusConstants.FaceSnapshotField.LOG_TYPE);
            sb.append(" == ");
            sb.append(searchType);
            sb.append(" && ");
        }
        // 时间范围
        if(timeScope) {
            String beginTime = faceSearchVO.getBeginTime();
            String endTime = faceSearchVO.getEndTime();
            long begin = 0L;
            long end = 0L;
            if (StrUtil.isNotEmpty(beginTime) && StrUtil.isNotEmpty(endTime)) {
                begin = DateUtil.parseDateTime(beginTime).getTime();
                end = DateUtil.parseDateTime(endTime).getTime();
            } else {
                // 默认查询范围 30天
                Date now = new Date();
                end = now.getTime();
                begin = DateUtil.offsetMonth(now, -1).getTime();
            }
            sb.append(MilvusConstants.FaceSnapshotField.TIMESTAMP);
            sb.append(" >= ");
            sb.append(begin);
            sb.append(" && ");
            sb.append(MilvusConstants.FaceSnapshotField.TIMESTAMP);
            sb.append(" <= ");
            sb.append(end);
            sb.append(" && ");
        }
        if(sb.length() > 0) {
            builder
                    .withExpr(sb.substring(0, sb.length() - 4));
        }
        return builder;
    }
}

附：虹软4.0特征向量转换方法

虹软4.0特征值 共2056个字节， 前8个字节固定是浮点数类型的2004,78（可能用于区别SDK的不同版本），后面的2048个字节是512个浮点数的且前256个是0。可以忽略。只需要把后256个特征向量复制出来就可以了。

参考：https://blog.csdn.net/Memory2414/article/details/116583985

/**
* 虹软byte[]特征值转List<Float>
* @param bytes
* @return
*/
public static List<Float> arcsoftToFloat(byte[] bytes) {
	List<Float> list = Lists.newArrayList();
	for (int i = 1024 + 8; i < bytes.length; i += 4) {
		byte[] bytes1 = {bytes[i], bytes[i + 1], bytes[i + 2], bytes[i + 3]};
		list.add(byte2float(bytes1, 0));
	}
	return list;
}
