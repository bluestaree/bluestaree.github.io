保留首尾字符，中间全部变为星号

  2021-01-05   笔记  评论 字数统计: 689字 阅读时长: 2分

因业务需要对手机号进行保密操作，对金额显示加上分隔符

代码如下：

public class StringReplaceUtils {

    /**
     * 对字符加星号处理：根据字符串长度替换
     * @param content 传入的字符串
     * @return 带星号的字符串
     */
    public static String replaceString2Star(String content) {
        int nameLength = content.length();
        if (nameLength >= 2 && nameLength <= 3) {
            return StringReplaceUtils.replaceString2Star(content, 1, 0);
        } else if (nameLength >= 4 && nameLength <= 6) {
            return StringReplaceUtils.replaceString2Star(content, 1, 1);
        } else if (nameLength == 7) {
            return StringReplaceUtils.replaceString2Star(content, 1, 2);
        } else if (nameLength == 8) {
            return StringReplaceUtils.replaceString2Star(content, 2, 2);
        } else if (nameLength == 9) {
            return StringReplaceUtils.replaceString2Star(content, 2, 3);
        } else if (nameLength == 10) {
            return StringReplaceUtils.replaceString2Star(content, 3, 3);
        } else if (nameLength >= 11) {
            return StringReplaceUtils.replaceString2Star(content, 3, 4);
        }
        return content;
    }

    /**
     * 对字符加星号处理：除前面几位和后面几位外，其他的字符以星号代替
     * @param content 传入的字符串
     * @param frontNum 保留前面字符的位数
     * @param endNum 保留后面字符的位数
     * @return 带星号的字符串
     */
    public static String replaceString2Star(String content, int frontNum, int endNum) {
        if (content == null || content.trim().isEmpty()) {
            return content;
        }

        int len = content.length();

        if (frontNum >= len || frontNum < 0 || endNum >= len || endNum < 0) {
            return content;
        }

        if (frontNum + endNum >= len) {
            return content;
        }


        int beginIndex = frontNum;
        int endIndex = len - endNum;
        char[] cardChar = content.toCharArray();

        for (int j = beginIndex; j < endIndex; j++) {
            cardChar[j] = '*';
        }

        return new String(cardChar);
    }

    /**
     * 手机号加密处理
     * @param phone 手机号
     * @return 带星号的字符串
     */
    public static String encryptPhone(String phone) {
        return replaceString2Star(phone, 3, 4);
    }

    /**
     * 判断字符串中是否包含中文
     * @param str 待校验字符串
     * @return 是否为中文
     * @warn 不能校验是否为中文标点符号
     */
    public static boolean isContainChinese(String str) {
        Pattern p = Pattern.compile("[\u4e00-\u9fa5]");
        Matcher m = p.matcher(str);
        if (m.find()) {
            return true;
        }
        return false;
    }

    /**
     * 将每三个数字（或字符）加上分隔符处理
     * 5000000.00 --> 5,000,000.00
     * 20000000 --> 20,000,000
     * @param str 000000101
     * @param separator 分隔符 "/"
     * @return 000/000/101
     */
    public static String strAddSeparator(String str, String separator) {
        if (str == null || str.trim().isEmpty() || separator == null || separator.trim().isEmpty()) {
            return str;
        }
        String addCommaStr = str; // 需要添加分隔符的字符串（整数）
        // 将传进数字反转
        String reverseStr = new StringBuilder(addCommaStr).reverse().toString();
        String strTemp = "";
        for (int i = 0; i < reverseStr.length(); i++) {
            if (i * 3 + 3 > reverseStr.length()) {
                strTemp += reverseStr.substring(i * 3, reverseStr.length());
                break;
            }
            strTemp += reverseStr.substring(i * 3, i * 3 + 3) + separator;
        }
        // 将 "000/000/001/" 中最后一个"/"去除
        if (strTemp.endsWith(separator)) {
            strTemp = strTemp.substring(0, strTemp.length() - 1);
        }
        // 将数字重新反转
        String resultStr = new StringBuilder(strTemp).reverse().toString();
        return resultStr;
    }
}
