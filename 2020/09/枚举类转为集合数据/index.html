枚举类转为集合数据

  2020-09-23   笔记  评论 字数统计: 332字 阅读时长: 1分

转载 · 原文链接：https://blog.csdn.net/weixin_43906974/article/details/107515953

场景：前端下拉框展示数据，老项目没有数据字典，在尽量不改动的情况下，可以使用以下方法

package com.example.demo.Enum;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * XXXX枚举
 */
public enum TimeKeepingEnum {

   FIRST(1,"20钻石/分钟"),
   SECOND(2,"30钻石/分钟"),
   THIRD(3,"40钻石/分钟");

   private int value;
   private String desc;

   private TimeKeepingEnum(int value, String desc) {
       this.value = value;
       this.desc = desc;

   }

    public int getValue() {
        return value;
    }

    public void setValue(int value) {
        this.value = value;
    }

    public String getDesc() {
        return desc;
    }

    public void setDesc(String desc) {
        this.desc = desc;
    }

    /**
     * 请求返回的Enum集合数据
     * @return
     */
    public static List<Map<String, Object>> getTimeKeepingEnumList() {
        List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
        for (TimeKeepingEnum timeKeepingEnum: TimeKeepingEnum.values()) {
            Map item= new HashMap<String, Object>();
            item.put("value",timeKeepingEnum.value);
            item.put("desc",timeKeepingEnum.desc);
            list.add(item);
        }
        return list;
    }

    public static void main(String[] args) {
        System.out.println(TimeKeepingEnum.getTimeKeepingEnumList());
    }
}
